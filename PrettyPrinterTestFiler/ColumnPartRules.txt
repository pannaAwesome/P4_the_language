IMPORT ../CsvFiler/users.csv IN users

MODEL user BEGIN
    COL nameRule BEGIN
        COL nameISFilled: COUNT(name) = 100
        COL heightISExpectedAverage: AVG(height) > 169 WHERE country = "Danmark"
