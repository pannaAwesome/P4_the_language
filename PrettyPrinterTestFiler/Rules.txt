MODEL user BEGIN
    nameRule: name IS LETTERS
    lastNameRule: lastName = "hej" OR lastName CONTAINS "goddag"
    heightRule: height < 200 AND height > 0
    balanceRule: balance = formerBalance + currentBalance - 5
